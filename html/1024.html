                  C {
                      #include
                      int main(){
                        printf("Hello, World !");
                        return 0;
                      }
                  }
              

                  C++ {
                      #include 
                      int main(){
                        std::cout << "Hello, World !";
                        return 0;
                      }                    
                  }
              

                  C# {
                    using System;
                    class Program
                    {
                        public static void Main(string[] args)
                        {
                            Console.WriteLine("Hello, World !");
                        }
                      }            
                  }
              

                  Java {
                      public class Main {
                        public static void main(String[] args) {
                            System.out.println("Hello, World !");
                        }
                    }                              
                  }

                    func main() {
                        fmt.Println("Hello, World!")
                    }
